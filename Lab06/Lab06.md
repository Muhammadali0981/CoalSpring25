## Q1
```asm
include irvine32.inc

	
.code
main proc
	mov ecx, 10
	mov eax, 0
	mov ebx, 1
	mov edx, 0
	L1: 
		call writeint
		call crlf
		mov edx,0

		add edx, eax
		add edx, ebx

		mov eax, ebx
		mov ebx, edx
	loop L1
	
	exit
main endp
end main
```

## Q2
```asm
include irvine32.inc
.data
	a DWORD 1
	
.code
main proc
	mov ecx, 4
	mov eax, 'I'
	
	L2:
		mov ebx, ecx
		mov ecx, a
		
		L1: 
			call writechar	
		loop L1

		call crlf
		inc a
		mov ecx, ebx
	loop L2

	exit
main endp
end main
```
